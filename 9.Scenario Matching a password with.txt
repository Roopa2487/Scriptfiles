9.Scenario: Matching a password with a user: username is "Rocky" pwd is Rocky@123 when both matches authentication is successful, if password is missed it should show invalid password, if username is invalid then invalid username


#!/bin/bash
echo "Enter the username"
read usr
echo "Enter the password"
read pw
if [ $usr == Rocky ] && [ $pw == Rocky@123 ]
then
        echo "Authentication successful"
elif [ $usr != Rocky ] && [ $pw == Rocky@123 ]
then
        echo "Invalid username"
elif [ $usr == Rocky ] && [ $pw != Rocky@123 ]
then
        echo "Invalid password"
else
        echo "Invalid password and username"
fi
