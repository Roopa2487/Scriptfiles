Print odd numbers from a range 1 to n

#!/bin/bash
echo "Enter the first value"
read n1
echo "Enter the second value"
read n2
while [ $n2 -ge $n1 ]
do
        mod=`expr $n2 % 2`
        if [ $mod -ne 0 ]
        then
                echo "$n2"
        fi
        n2=`expr $n2 - 1`
done
